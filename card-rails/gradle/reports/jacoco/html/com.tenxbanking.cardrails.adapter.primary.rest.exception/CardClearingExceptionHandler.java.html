package com.tenxbanking.cardrails.adapter.primary.rest.exception;

import com.tenxbanking.cardrails.adapter.primary.rest.model.response.ErrorResponse;
import com.tenxbanking.cardrails.adapter.secondary.cardclearing.exception.CardClearingSaveCardAuthException;
import com.tenxbanking.cardrails.domain.exception.CardClearingBusinessException;
import com.tenxbanking.cardrails.domain.exception.CardClearingDebitCardManagementException;
import com.tenxbanking.cardrails.domain.exception.CardClearingFeeAndChargesException;
import com.tenxbanking.cardrails.domain.exception.CardClearingSubscriptionNotFoundException;
import org.springframework.http.HttpStatus;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.ResponseStatus;

public class CardClearingExceptionHandler {

  @ExceptionHandler(CardClearingFeeAndChargesException.class)
  @ResponseStatus(HttpStatus.UNPROCESSABLE_ENTITY)
  public ErrorResponse handleCardClearingFeeAndChargesUnsupportedException(
      CardClearingFeeAndChargesException e) {
    return new ErrorResponse(e.getMessage());
  }

  @ExceptionHandler(CardClearingSaveCardAuthException.class)
  @ResponseStatus(HttpStatus.UNPROCESSABLE_ENTITY)
  public ErrorResponse handleCardClearingSaveCardAuthUnsupportedException(
      CardClearingSaveCardAuthException e) {
    return new ErrorResponse(e.getMessage());
  }

  @ExceptionHandler(CardClearingBusinessException.class)
  @ResponseStatus(HttpStatus.UNPROCESSABLE_ENTITY)
  public ErrorResponse handleCardClearingTransactionMessageException(
      CardClearingBusinessException e) {
    return new ErrorResponse(e.getMessage());
  }

  @ExceptionHandler(CardClearingDebitCardManagementException.class)
  @ResponseStatus(HttpStatus.UNPROCESSABLE_ENTITY)
  public ErrorResponse handleCardClearingDebitCardManagementMessageException(
      CardClearingDebitCardManagementException e) {
    return new ErrorResponse(e.getMessage());
  }

  @ExceptionHandler(CardClearingDebitCardManagementException.class)
  @ResponseStatus(HttpStatus.UNPROCESSABLE_ENTITY)
  public ErrorResponse handleCardClearingTransactionMessageParseException(
      CardClearingDebitCardManagementException e) {
    return new ErrorResponse(e.getMessage());
  }

  @ExceptionHandler(CardClearingSubscriptionNotFoundException.class)
  @ResponseStatus(HttpStatus.UNPROCESSABLE_ENTITY)
  public ErrorResponse handleCardClearingSubscriptionNotFoundException(
      CardClearingSubscriptionNotFoundException e) {
    return new ErrorResponse(e.getMessage());
  }


}
