package com.tenxbanking.cardrails.domain.model.card;

import com.tenxbanking.cardrails.adapter.secondary.database.model.MessageTypeEnum;
import java.math.BigDecimal;
import java.util.UUID;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Data;

@Data
@AllArgsConstructor
@Builder(toBuilder = true)
public class Transaction {

  private UUID id;
  private UUID correlationId;
  private String transactionDate;
  private String retrievalReferenceNumber;
  private Merchant merchant;
  private String cardId;
  private BigDecimal amount;
  private String currency;
  private String qualifier;
  private String authenticationCode;
  private int processingCode;
  private Pos pos;
  private int functionCode;
  private String transactionLifeCycleID;
  private String iccData;
  private String mappingServiceAccountNumber;
  private String dataRecord;
  private int messageNumber;
  private Billing billing;
  private BigDecimal conversionRate;
  private Enum<MessageTypeEnum> messageType;

  public Transaction addTransactionIds(UUID transactionId, UUID correlationId) {
    return this.toBuilder().id(transactionId).correlationId(correlationId).build();
  }
}
