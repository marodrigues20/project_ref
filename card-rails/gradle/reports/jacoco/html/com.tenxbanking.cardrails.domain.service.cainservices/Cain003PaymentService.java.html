package com.tenxbanking.cardrails.domain.service.cainservices;


import com.tenx.universalbanking.transactionmessage.TransactionMessage;
import com.tenxbanking.cardrails.domain.model.messages.Cain003;
import com.tenxbanking.cardrails.domain.port.CainPaymentService;
import com.tenxbanking.cardrails.domain.service.message.builders.Cain003MessageBuilder;
import java.text.ParseException;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

@Service(value = "cain003PaymentService")
public class Cain003PaymentService implements CainPaymentService {

  @Autowired
  private Cain003MessageBuilder cain003MessageBuilder;

  @Override
  public TransactionMessage buildMessage(Cain003 cain003)
      throws ParseException {
    TransactionMessage transactionMessage = new TransactionMessage();
    transactionMessage = cain003MessageBuilder.setTransactionMessageHeader(transactionMessage);
    transactionMessage = cain003MessageBuilder.setTransactionMessage(transactionMessage,cain003);
    return transactionMessage;
  }
}
